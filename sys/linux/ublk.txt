define UBLK_PARAM_TYPE_BASIC	(1 << 0)
define UBLK_PARAM_TYPE_DISCARD	(1 << 1)
define UBLK_PARAM_TYPE_DEVT	(1 << 2)
define UBLK_PARAM_TYPE_ZONED	(1 << 3)
ublk_param_types = UBLK_PARAM_TYPE_BASIC, UBLK_PARAM_TYPE_DISCARD, UBLK_PARAM_TYPE_DEVT, UBLK_PARAM_TYPE_ZONED

ublk_params {
	len	len[parent, int32]
	types	flags[ublk_param_types, int32]
	basic	ublk_param_basic
	discard	ublk_param_discard
	devt	ublk_param_devt
	zoned	ublk_param_zoned
}

define UBLK_ATTR_READ_ONLY	(1 << 0)
define UBLK_ATTR_ROTATIONAL	(1 << 1)
define UBLK_ATTR_VOLATILE_CACHE	(1 << 2)
define UBLK_ATTR_FUA	(1 << 3)
ublk_attr_flags = UBLK_ATTR_READ_ONLY, UBLK_ATTR_ROTATIONAL, UBLK_ATTR_VOLATILE_CACHE, UBLK_ATTR_FUA

ublk_param_basic {
	attrs			flags[ublk_attr_flags, int32]
	logical_bs_shift	int8[9:12]
	physical_bs_shift	int8[9:12]
	io_opt_shift		int8
	io_min_shift		int8
	max_sectors		int32
	chunk_sectors		int32
	dev_sectors		int64
	virt_boundary_mask	int64
}

ublk_param_discard {
	discard_alignment		int32
	discard_granularity		int32
	max_discard_sectors		int32
	max_write_zeroes_sectors	int32
	max_discard_segments		int16
	reserved0			const[0, int16]
}

ublk_param_devt {
	char_major	int32
	char_minor	int32
	disk_major	int32
	disk_minor	int32
}

	max_open_zones		int32
	max_active_zones	int32
	max_zone_append_sectors	int32
	reserved		array[const[0, int8], 20]
}
