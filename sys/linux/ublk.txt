# --------------------------------------- Data types for ublk ctrl commands ---------------------------------------
type ublk_ctrl_cmd[DEV_ID, LEN, ADDR, DATA] {
	dev_id		DEV_ID
	queue_id	const[-1, int16]

	len		LEN
	addr		ADDR

	data		DATA

	dev_path_len	const[0, int16]
	pad		const[0, int16]
	reserved	const[0, int32]

	pad2_unused	array[const[0, int32], 12]
}

type ublk_ctrl_cmd$add_dev ublk_ctrl_cmd[const[-1, int32], len[addr, int16], ptr64[in, ublk_ctrl_dev_info_new], const[0, int64]]
type ublk_ctrl_cmd$set_params ublk_ctrl_cmd[ublk_dev_id, len[addr, int16], ptr64[in, ublk_params], const[0, int64]]
type ublk_ctrl_cmd$start_dev ublk_ctrl_cmd[ublk_dev_id, const[0, int16], const[0, int64], pid]
type ublk_ctrl_cmd$stop_dev ublk_ctrl_cmd[ublk_dev_id, const[0, int16], const[0, int64], const[0, int64]]
type ublk_ctrl_cmd$start_user_recovery ublk_ctrl_cmd[ublk_dev_id, const[0, int16], const[0, int64], const[0, int64]]
type ublk_ctrl_cmd$end_user_recovery ublk_ctrl_cmd[ublk_dev_id, const[0, int16], const[0, int64], pid]
type ublk_ctrl_cmd$del_dev ublk_ctrl_cmd[ublk_dev_id, const[0, int16], const[0, int64], const[0, int64]]

type ublk_ctrl_cmd$get_queue_affinity ublk_ctrl_cmd[ublk_dev_id, len[addr, int16], ptr64[out, int64], int64[0:UBLK_MAX_HW_QUEUES]]
type ublk_ctrl_cmd$get_device_info ublk_ctrl_cmd[ublk_dev_id, len[addr, int16], ptr64[out, ublk_ctrl_dev_info_all], const[0, int64]]
type ublk_ctrl_cmd$get_params ublk_ctrl_cmd[ublk_dev_id, len[addr, int16], ptr64[out, ublk_params], const[0, int64]]
type ublk_ctrl_cmd$get_features ublk_ctrl_cmd[ublk_dev_id, len[addr, int16], ptr64[out, int64], const[0, int64]]

type ublk_ctrl_cmd$any ublk_ctrl_cmd[int32, int16, int64, int64]

type ublk_ctrl_dev_info[DEV_ID] {
	nr_hw_queues		int16[1:UBLK_MAX_HW_QUEUES]
	queue_depth		int16[1:UBLK_MAX_QUEUE_DEPTH]
	state			const[0, int16]
	pad0			const[0, int16]
	max_io_buf_bytes	const[UBLK_MAX_IO_BYTES, int32]
	dev_id			DEV_ID
	ublk_pid		pid
	pad1			const[0, int32]
	flags			flags[ublk_f_flags, int64]
	ublk_flags		const[0, int64]
	owner_uid		const[0, int32]
	owner_gid		const[0, int32]
	reserved1		const[0, int64]
	reserved2		const[0, int64]
}

ublk_f_flags = UBLK_F_SUPPORT_ZERO_COPY, UBLK_F_URING_CMD_COMP_IN_TASK, UBLK_F_NEED_GET_DATA, UBLK_F_USER_RECOVERY, UBLK_F_USER_RECOVERY_REISSUE, UBLK_F_UNPRIVILEGED_DEV, UBLK_F_CMD_IOCTL_ENCODE, UBLK_F_USER_COPY, UBLK_F_ZONED

ublk_ctrl_dev_info_new [
	new_dev	ublk_ctrl_dev_info[const[-1, int32]]
	any_dev	ublk_ctrl_dev_info[int32]
]

ublk_ctrl_dev_info_all [
	ublk_ctrl_dev_info_new		ublk_ctrl_dev_info_new
	ublk_ctrl_dev_info_existing	ublk_ctrl_dev_info[ublk_dev_id]
]

define UBLK_PARAM_TYPE_BASIC	(1 << 0)
define UBLK_PARAM_TYPE_DISCARD	(1 << 1)
define UBLK_PARAM_TYPE_DEVT	(1 << 2)
define UBLK_PARAM_TYPE_ZONED	(1 << 3)
ublk_param_types = UBLK_PARAM_TYPE_BASIC, UBLK_PARAM_TYPE_DISCARD, UBLK_PARAM_TYPE_DEVT, UBLK_PARAM_TYPE_ZONED

ublk_params {
	len	len[parent, int32]
	types	flags[ublk_param_types, int32]
	basic	ublk_param_basic
	discard	ublk_param_discard
	devt	ublk_param_devt
	zoned	ublk_param_zoned
}

define UBLK_ATTR_READ_ONLY	(1 << 0)
define UBLK_ATTR_ROTATIONAL	(1 << 1)
define UBLK_ATTR_VOLATILE_CACHE	(1 << 2)
define UBLK_ATTR_FUA	(1 << 3)
ublk_attr_flags = UBLK_ATTR_READ_ONLY, UBLK_ATTR_ROTATIONAL, UBLK_ATTR_VOLATILE_CACHE, UBLK_ATTR_FUA

ublk_param_basic {
	attrs			flags[ublk_attr_flags, int32]
	logical_bs_shift	int8[9:12]
	physical_bs_shift	int8[9:12]
	io_opt_shift		int8
	io_min_shift		int8
	max_sectors		int32
	chunk_sectors		int32
	dev_sectors		int64
	virt_boundary_mask	int64
}

ublk_param_discard {
	discard_alignment		int32
	discard_granularity		int32
	max_discard_sectors		int32
	max_write_zeroes_sectors	int32
	max_discard_segments		int16
	reserved0			const[0, int16]
}

ublk_param_devt {
	char_major	int32
	char_minor	int32
	disk_major	int32
	disk_minor	int32
}

ublk_param_zoned {
	max_open_zones		int32
	max_active_zones	int32
	max_zone_append_sectors	int32
	reserved		array[const[0, int8], 20]
}
